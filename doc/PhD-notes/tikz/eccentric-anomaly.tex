\begin{tikzpicture}
  \def\xstart{0}
  \def\ystart{0}
  \def\r{4}
  \def\a{4}
  \def\e{0.6}
  % geometry of ellipse
  \pgfmathsetmacro\b{{\a * sqrt(1.0 - \e * \e)}}
  \pgfmathsetmacro\c{{sqrt(\a * \a - \b * \b)}}
  % coordinates of centers
  \pgfmathsetmacro\xc{{\xstart - \r *cos(0)}}
  \pgfmathsetmacro\yc{{\ystart - \r *sin(0)}}
  % focal point (aka earth)
  \pgfmathsetmacro\xf{\xc + \c }
  \pgfmathsetmacro\yf{\yc }
  % projection of P on x-axis
  \pgfmathsetmacro\xp{{\xf + \c / 3e0}}
  \pgfmathsetmacro\ype{{sqrt((\a * \a - (\xp -\xc ) * (\xp -\xc ))*(1.0 - \e *\e )) + \yc }}
  \pgfmathsetmacro\ypc{{sqrt(\r *\r - (\xp -\xc ) * (\xp -\xc )) + \yc }}

  % auxiliary circle arc
  \draw ( \xstart , \ystart ) arc(0:180:\r);
  \node[] (perigee) 
    at ( \xstart , \ystart )
    {};
  \node[anchor=north] () at (perigee){Perigee};
  
  \node[] (center) at (\xc , \yc ) {};
  %\node[anchor=north] () at (center){Center};

  \node[] (apogee) 
    at ({\xstart - \r *cos(0)+ \r *cos(180)}, {\ystart - \r *sin(0)+ \r *sin(180)})
    {};
  \node[anchor=north] () at (apogee){Apogee};
  
  % ellipse arc
  \draw[red] ( \xstart , \ystart ) arc(0:180:\a cm and \b cm);
  
  % earth (one foci)
  \node[] (earth) 
    at ( \xf , \yf ){};
  \node[anchor=north] () 
    at (earth){$\bm{\oplus}$};

  % point on ellipse and auxiliary sphere
  \draw[] (\xp , \yc ) -- (\xp , \ypc );

  \draw[] ( \xc , \yc ) -- (\xp , \ypc );
  \node[] (pcircle) at (\xp , \ypc ){};
  \pic [draw, ->, "$E$", angle eccentricity=1.5] {angle = earth--center--pcircle};
  \draw[->] (\xf , \yf )  -- (\xp , \ype );
  \node[] (ecircle) at (\xp , \ype ){};
  \pic [draw, ->, "v", angle eccentricity=1.2] {angle = perigee--earth--ecircle};
    
  % x-axis
  \draw[] ($(apogee) - (.2,0)$)  -- ($(perigee)+(.2,0)$);

  % y-axis
  \draw[] ($(center)+(0, -.2)$) -- ($(center)+(0,\r+.2)$);

\end{tikzpicture}
